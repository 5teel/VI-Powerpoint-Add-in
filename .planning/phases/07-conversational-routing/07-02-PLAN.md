    export function responseClassToRoute(
      cls: ResponseClass,
      ctx: { isFirstTurn: boolean; hasToolCall: boolean; plannedSlideCount?: number }
    ): { route: ResponseRoute; degraded?: true } {
      switch (cls) {
        case "clarify":
        case "refuse":
          return { route: "narrative-conversational" };
        case "data":
          if (!ctx.hasToolCall) return { route: "narrative" };
          return ctx.plannedSlideCount && ctx.plannedSlideCount > 1
            ? { route: "section-plan" }
            : { route: "new-composition" };
        case "section":
          return ctx.hasToolCall ? { route: "section-plan" } : { route: "narrative" };
        case "modify":
        case "variant":
          if (ctx.isFirstTurn) {
            return ctx.hasToolCall
              ? { route: "new-composition", degraded: true }
              : { route: "narrative", degraded: true };
          }
          return { route: "refinement" };
      }

    export function streamAndClassify(
      question: string,
      ctx: DispatchContext,
      cb: DispatchCallbacks
    ): AbortController {
      const outer = new AbortController();
      let capturedToolCall: CubeSqlApiToolCall | null = null;
      let capturedCommentary = "";
      let chatId: string | null = ctx.chatId;
      let inner: AbortController | null = null;

      // Forward outer abort → inner stream abort.
      outer.signal.addEventListener("abort", () => {
        try { inner?.abort(); } catch { /* ignore */ }
      });

      inner = streamCubeAI(question, ctx.chatId, {
        onPhaseChange: cb.onPhaseChange,
        onContent: (content) => {
          capturedCommentary = content;
          cb.onContent(content);
        },
        onToolCall: (tc) => {
          capturedToolCall = tc;
          cb.onToolCall?.(tc);
        },
        onError: cb.onError,
        onComplete: async (result: CubeAIStreamResult) => {
          if (outer.signal.aborted) return;
          chatId = result.chatId ?? chatId;

          const isFirstTurn = !ctx.hasLastBuild;
          const hasToolCall = capturedToolCall !== null;

          // STAGE 1
          const stage1Start = performance.now();
          const classification = await classifyResponse(
            {
              assistantText: result.content || capturedCommentary || "",
              toolCallPresent: hasToolCall,
              lastSlideTitle: ctx.lastSlideTitle,
              userQuestion: ctx.userQuestion,
              isFirstTurn,
            },
            outer.signal
          );
          if (outer.signal.aborted) return;
          const stage1Latency = Math.round(performance.now() - stage1Start);

          // D-03 — invoke planSection ONLY on class=section
          let sectionPlan: SectionPlan | undefined;
          let plannedSlideCount: number | undefined;
          if (classification.class === "section" && hasToolCall && capturedToolCall) {
            try {
              sectionPlan = await new Promise<SectionPlan>((resolve, reject) => {
                let settled = false;
                const tc = capturedToolCall!;
                const input: MetaComposerInput = {
                  userQuestion: ctx.userQuestion,
                  cubeMeta: {
                    queryTitle: tc.input.queryTitle ?? "",
                    description: tc.input.description ?? "",
                    commentary: capturedCommentary || (result.content ?? ""),
                    chartCategory: tc.input.chartCategory,
                  },
                  signal: outer.signal,
                };
                planSection(input, {
                  onPartialPlan: () => {},
                  onFinal: (p) => { if (!settled) { settled = true; resolve(p); } },
                  onError: (e) => { if (!settled) { settled = true; reject(e); } },
                });
              });
              plannedSlideCount = sectionPlan.slides.length;
            } catch (err) {
              if (outer.signal.aborted) return;
              logEvent("planSection.failed_defaulting_single", { err: String(err), surface: ctx.surface });
              sectionPlan = undefined;
              plannedSlideCount = 1;
            }
          }

          // Compute route
          const { route: baseRoute, degraded } = responseClassToRoute(
            classification.class,
            { isFirstTurn, hasToolCall, plannedSlideCount }
          );
          let route: ResponseRoute = baseRoute;

          // D-08 STAGE 2 — modify/variant on continuation
          let refinementPath: "composer-only" | "cube-ai+composer" | undefined;
          if (
            (classification.class === "modify" || classification.class === "variant") &&
            !isFirstTurn &&
            hasToolCall
          ) {
            const stage2 = await classifyRefinement(
              ctx.userQuestion,
              ctx.lastSlideTitle ?? "",
              outer.signal
            );
            if (outer.signal.aborted) return;
            refinementPath = stage2.path;
          }

          // R7 — single canonical telemetry emission
          logEvent("response_classified", {
            class: classification.class,
            rationale: classification.rationale,
            latencyMs: stage1Latency,
            route,
            surface: ctx.surface,
            ...(degraded ? { degraded: true } : {}),
            ...(classification.rationale === "classifier-fallback" ? { fallback: true } : {}),
            ...(refinementPath ? { refinementPath } : {}),
            ...(plannedSlideCount !== undefined ? { plannedSlideCount } : {}),
          });

          await cb.onClassified({
            class: classification.class,
            route,
            rationale: classification.rationale,
            toolCall: capturedToolCall,
            commentary: capturedCommentary || (result.content ?? ""),
            chatId,
            refinementPath,
            sectionPlan,
            degraded,
            latencyMs: stage1Latency,
          });
        },
      });

      return outer;
    }
