    const handleSubmit = useCallback(
      (question: string) => {
        if (!question.trim() || phase !== null) return;

        setLastQuestion(question);
        setMessages((prev) => [...prev, { role: "user", content: question }]);
        setQuery("");
        setStreamingContent("");

        const wrappedQuestion = buildSlidePrompt(question);
        const submitLastSlideTitle = lastSlideTitle;

        const controller = streamAndClassify(
          wrappedQuestion,
          {
            surface: "chat",
            chatId,
            lastSlideTitle: submitLastSlideTitle ?? undefined,
            hasLastBuild: lastBuildRef.current !== null,
            userQuestion: question,
          },
          {
            onPhaseChange: (p: StreamPhase) => setPhase(p),
            onContent: (content: string) => setStreamingContent(content),
            onToolCall: () => { /* dispatcher captures toolCall internally */ },
            onError: (error: CubeAIError) => {
              setMessages((prev) => [...prev, { role: "error", content: error.message, error }]);
              setStreamingContent("");
              setPhase(null);
            },
            onClassified: async ({
              class: cls,
              route,
              toolCall,
              commentary,
              chatId: newChatId,
              refinementPath,
              sectionPlan,
              degraded,
            }) => {
              // 1. Append assistant message with responseClass populated.
              const raw = commentary || "";
              const parsed = extractSlideContent(raw);
              const displayText = parsed
                ? summarizeSlideContent(parsed)
                : raw || "(No response received)";

              let newMsgIdx = -1;
              setMessages((prev) => {
                newMsgIdx = prev.length;
                return [
                  ...prev,
                  {
                    role: "assistant",
                    content: displayText,
                    rawContent: raw,
                    slideState: "idle",
                    toolCall: toolCall ?? undefined,
                    commentary: commentary || undefined,
                    responseClass: cls,
                  },
                ];
              });
              setStreamingContent("");
              setPhase(null);
              if (newChatId) setChatId(newChatId);

              // 2. Telemetry annotation for D-06 first-turn degrade analytics.
              if (degraded) {
                logEvent("first_turn_degraded", { class: cls, surface: "chat" });
              }

              // 3. Dispatch on route.
              switch (route) {
                case "narrative-conversational":
                case "narrative":
                  // Plain text bubble; no further action.
                  return;

                case "refinement": {
                  const last = lastBuildRef.current;
                  if (!last || !toolCall) return; // safety — should not happen on continuation
                  if (refinementPath === "composer-only") {
                    await runCompositionForRefinement(newMsgIdx, {
                      userQuestion: `${last.userQuestion}\n\nREFINEMENT: ${question}`,
                      toolCall: last.toolCall,
                      rows: last.rows,
                      commentary: last.commentary,
                    });
                  } else {
                    // cube-ai+composer (default refinement path).
                    await runCompositionForRefinement(newMsgIdx, {
                      userQuestion: question,
                      toolCall,
                      rows: [],
                      commentary,
                    });
                  }
                  return;
                }

                case "section-plan": {
                  if (!toolCall || !sectionPlan) return;
                  try {
                    const cubeQuery = translateSql(toolCall.input.sqlQuery);
                    const cubeResp = await loadCubeData(cubeQuery, { signal: controller.signal });
                    setSectionBuild({
                      messageIndex: newMsgIdx,
                      input: {
                        userQuestion: question,
                        toolCall,
                        cubeRows: cubeResp.data,
                        commentary,
                      },
                    });
                  } catch (err) {
                    const msg = err instanceof Error ? err.message : String(err);
                    setMessages((prev) => [
                      ...prev,
                      { role: "error", content: `Couldn't fetch section data: ${msg}` },
                    ]);
                  }
                  return;
                }

                case "new-composition":
                  // Existing Phase 5 path — message sits idle until user clicks Create Slide.
                  return;
              }
            },
          }
        );

        controllerRef.current = controller;
      },
      [phase, chatId, lastSlideTitle, runCompositionForRefinement]
    );
