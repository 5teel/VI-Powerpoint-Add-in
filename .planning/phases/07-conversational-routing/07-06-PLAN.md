    ```ts
    /**
     * Phase 7 R6 telemetry dump helper.
     *
     * Reads a localStorage dump (JSON array of TelemetryEvent), filters to
     * "response_classified" events, and prints a summary including p95 latency.
     *
     * Usage:
     *   1. In the deployed Railway build, open DevTools.
     *   2. Run: copy(localStorage.getItem("summit.ai.trace.v1"))
     *   3. Paste into ./classifier-dump.json (overwrite).
     *   4. Run: npx tsx scripts/dump-classifier-telemetry.ts ./classifier-dump.json
     */
    import { readFileSync } from "node:fs";

    interface TelemetryEvent {
      t: number;
      name: string;
      payload?: Record<string, unknown>;
    }

    function p95(values: number[]): number | null {
      if (values.length === 0) return null;
      const sorted = [...values].sort((a, b) => a - b);
      const idx = Math.min(sorted.length - 1, Math.ceil(0.95 * sorted.length) - 1);
      return sorted[idx];
    }

    function bumpCount(map: Map<string, number>, key: string): void {
      map.set(key, (map.get(key) ?? 0) + 1);
    }

    function main(): void {
      const path = process.argv[2];
      if (!path) {
        console.error("Usage: npx tsx scripts/dump-classifier-telemetry.ts <dump-file>");
        process.exit(2);
      }

      const raw = readFileSync(path, "utf8");
      const events = JSON.parse(raw) as TelemetryEvent[];
      const classified = events.filter((e) => e.name === "response_classified");

      if (classified.length === 0) {
        console.log("No `response_classified` events found in dump.");
        process.exit(0);
      }

      const latencies = classified
        .map((e) => Number(e.payload?.latencyMs))
        .filter((n) => Number.isFinite(n));

      const byClass = new Map<string, number>();
      const byRoute = new Map<string, number>();
      const bySurface = new Map<string, number>();
      let degradedCount = 0;
      let fallbackCount = 0;

      for (const e of classified) {
        const p = e.payload ?? {};
        bumpCount(byClass, String(p.class ?? "(unknown)"));
        bumpCount(byRoute, String(p.route ?? "(unknown)"));
        bumpCount(bySurface, String(p.surface ?? "(unknown)"));
        if (p.degraded === true) degradedCount += 1;
        if (p.fallback === true) fallbackCount += 1;
      }

      const fmt = (m: Map<string, number>) =>
        [...m.entries()]
          .sort((a, b) => b[1] - a[1])
          .map(([k, v]) => `  ${k}: ${v}`)
          .join("\n");

      const p95Val = p95(latencies);

      console.log("─── Phase 7 classifier telemetry summary ───");
      console.log(`Total response_classified events: ${classified.length}`);
      console.log(`Latency samples: ${latencies.length}`);
      if (p95Val !== null) {
        console.log(`Latency p95: ${p95Val}ms (R6 target: ≤ 600ms — ${p95Val <= 600 ? "PASS" : "FAIL"})`);
        const min = Math.min(...latencies);
        const max = Math.max(...latencies);
        const avg = latencies.reduce((a, b) => a + b, 0) / latencies.length;
        console.log(`Latency min/avg/max: ${min}ms / ${avg.toFixed(0)}ms / ${max}ms`);
      }
      console.log("\nBy class:");
      console.log(fmt(byClass));
      console.log("\nBy route:");
      console.log(fmt(byRoute));
      console.log("\nBy surface:");
      console.log(fmt(bySurface));
      console.log(`\nDegraded (D-05 first-turn modify/variant): ${degradedCount}`);
      console.log(`Fallback (R9 classifier error): ${fallbackCount}`);
    }

    main();
    ```
