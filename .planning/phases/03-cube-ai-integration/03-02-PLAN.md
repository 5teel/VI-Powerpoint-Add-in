```typescript
export type StreamPhase = "connecting" | "connected" | "streaming" | "complete";

export interface CubeAIStreamResult {

}

  message: string;
  type: "cors" | "auth" | "network" | "server" | "timeout" | "empty" | "unknown";

}

export interface StreamCallbacks {
  onPhaseChange: (phase: StreamPhase) => void;
  onContent: (accumulatedContent: string) => void;
  onComplete: (result: CubeAIStreamResult) => void;
  onError: (error: CubeAIError) => void;
}

export function streamCubeAI(
  question: string,
  chatId: string | null,
  callbacks: StreamCallbacks
): AbortController;
```

```typescript
  role: "user" | "assistant" | "error";

  connecting: "Connecting to Cube AI...",
  connected: "Analyzing your question...",
  streaming: "Generating response...",
  complete: "",

```

**State variables:**

```typescript
const [messages, setMessages] = useState<ChatMessage[]>([]);
const [streamingContent, setStreamingContent] = useState<string>("");
const [phase, setPhase] = useState<StreamPhase | null>(null);
const [query, setQuery] = useState("");
const [lastQuestion, setLastQuestion] = useState<string>("");
const [chatId, setChatId] = useState<string | null>(null);
const controllerRef = useRef<AbortController | null>(null);
```

**useEffect cleanup** (Pitfall 2 — abort on unmount):

```typescript
useEffect(() => {
  return () => { controllerRef.current?.abort(); };
}, []);
```

**handleSubmit function:**
1. Guard: if `!question.trim()` or `phase` is not null (already streaming), return early.
2. Accept `question` parameter (string) so retry can pass the same question.
3. Set `lastQuestion` to the question text.
4. Add user message to `messages`: `{ role: "user", content: question }`
5. Clear `query` input, set `streamingContent` to `""`.
6. Call `streamCubeAI(question, chatId, callbacks)` and store the returned AbortController in `controllerRef.current`.

**Callbacks passed to streamCubeAI:**
- `onPhaseChange: (p) => setPhase(p)` -- drives spinner display
- `onContent: (content) => setStreamingContent(content)` -- progressive text
- `onComplete: (result) => { setMessages(prev => [...prev, { role: "assistant", content: result.content }]); setStreamingContent(""); setPhase(null); if (result.chatId) setChatId(result.chatId); }`
- `onError: (error) => { setMessages(prev => [...prev, { role: "error", content: error.message, error }]); setStreamingContent(""); setPhase(null); }`

**handleRetry function** (per D-05):
```typescript
const handleRetry = () => { if (lastQuestion) handleSubmit(lastQuestion); };
```

**handleKeyDown:** Same as current (Enter without Shift submits).
