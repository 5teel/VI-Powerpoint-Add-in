```typescript
export interface TextOnlyContent {
  type: "text-only";
  title: string;
  bullets: string[];
  insight: string;
}

export interface ChartTextContent {
  type: "chart-text";
  title: string;
  chartImageBase64?: string;
  summaryBullets: string[];
  insight: string;
}

export interface TableTextContent {
  type: "table-text";
  title: string;
  headers: string[];
  rows: (string | number)[][];
  summary: string;
}

export interface FullCombinationContent {
  type: "full-combination";
  title: string;
  chartImageBase64?: string;
  headers: string[];
  rows: (string | number)[][];
  insight: string;
}

export type SlideContent =
  | TextOnlyContent
  | ChartTextContent
  | TableTextContent
  | FullCombinationContent;
```

    - extractSlideContent with valid table-text JSON returns TableTextContent with headers and rows
    - extractSlideContent with valid chart-text JSON returns ChartTextContent (chartImageBase64 stripped)
    - extractSlideContent with valid full-combination JSON returns FullCombinationContent (chartImageBase64 stripped)
    - extractSlideContent with JSON inside markdown fences (```json ... ```) extracts and parses correctly
    - extractSlideContent with JSON embedded in commentary text ("Here is the data: {...} Hope this helps") extracts correctly
    - extractSlideContent with completely invalid input (no JSON at all) returns null
    - extractSlideContent with valid JSON but missing required fields (no title) returns null
    - extractSlideContent with unrecognized type field (e.g., "type": "dashboard") returns null (per D-08, LYOT-01)
    - extractSlideContent always deletes chartImageBase64 from parsed result (Pitfall 3)
    - fallbackToTextOnly converts multi-line text into TextOnlyContent with lines as bullets
    - fallbackToTextOnly strips markdown bullet markers (- and *) from lines
    - fallbackToTextOnly caps bullets at 6, puts overflow in insight
    - fallbackToTextOnly with empty string returns a valid TextOnlyContent with default bullet
  </behavior>
  <action>
    Write tests first following the project pattern in cubeai.test.ts (vitest, describe/it/expect).

    Then implement src/taskpane/services/schemaParser.ts with:
    - `extractSlideContent(raw: string): SlideContent | null` — multi-stage extraction per D-04:
      Stage 1: Try JSON.parse on trimmed input (handles clean JSON responses)
      Stage 2: Regex extract from markdown code fences `/```(?:json)?\s*([\s\S]*?)```/`
      Stage 3: Find first `{` to last `}` and try parsing
      Stage 4: Return null (caller uses fallbackToTextOnly per D-03)
    - After successful parse, validate with `validateSlideContent(obj)`:
      Check `type` is one of: "text-only", "chart-text", "table-text", "full-combination"
      Check `title` is non-empty string
      Per type, check required fields exist with correct types (bullets is array, headers is array, rows is array of arrays, insight/summary is string)
      Always `delete` chartImageBase64 from the parsed object (per Pitfall 3 — Cube AI may hallucinate base64)
      Return null for any validation failure
    - `fallbackToTextOnly(rawContent: string): TextOnlyContent` — per D-03, D-08:
      Split on newlines, strip markdown bullet markers (`/^[-*]\s*/`), filter empty lines
      Cap at 6 bullets, overflow goes to insight field
      Title defaults to "AI Insights"
      If no lines at all, return `{ type: "text-only", title: "AI Insights", bullets: ["No structured data available"], insight: "" }`

    Import types from `../slide/types` — use `SlideContent`, `TextOnlyContent`.
    Export both functions.
